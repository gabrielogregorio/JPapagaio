# BANCO
CREATE DATABASE TASKSNOW;

# TABELA
CREATE TABLE TAREFAS 
(
	ID INT PRIMARY KEY AUTO_INCREMENT,
	TITULO VARCHAR(50),
	DESCRICAO VARCHAR(255),
	PRIORIDADE INT,
	STATUS VARCHAR(20)
)
